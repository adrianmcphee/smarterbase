Distributed Locks

Race-Free Updates

Redis distributed locks eliminate S3 race conditions. True atomic operations across multiple servers.

  • Prevents lost updates
  • Automatic retry with backoff
  • TTL protection against deadlocks
  • Production-tested at scale

Quick Start

Get started in minutes with Go 1.18+

1

Installation

go get github.com/adrianmcphee/smarterbase
2

Basic Setup

package main

import (
    "context"
    "github.com/adrianmcphee/smarterbase"
)

func main() {
    // Development: Filesystem backend
    backend := smarterbase.NewFilesystemBackend("./data")
    defer backend.Close()

    // Production: S3 with Redis locks
    // backend := smarterbase.NewS3BackendWithRedisLock(
    //     s3Client, "my-bucket", redisClient,
    // )

    store := smarterbase.NewStore(backend)
    ctx := context.Background()

    // Create
    user := &User{
        ID:    smarterbase.NewID(),
        Email: "alice@example.com",
        Name:  "Alice",
    }
    store.PutJSON(ctx, "users/"+user.ID, user)

    // Read
    var retrieved User
    store.GetJSON(ctx, "users/"+user.ID, &retrieved)
}
3

Add Indexing

// Setup Redis indexer
redisClient := redis.NewClient(&redis.Options{
    Addr: "localhost:6379",
})
redisIndexer := smarterbase.NewRedisIndexer(redisClient)

// Register index
redisIndexer.RegisterMultiValueIndex(
    "users", "email",
    func(data []byte) (string, string, error) {
        var user User
        json.Unmarshal(data, &user)
        return user.ID, user.Email, nil
    },
)

// Create with automatic indexing
indexManager := smarterbase.NewIndexManager(store).
    WithRedisIndexer(redisIndexer)

indexManager.Create(ctx, "users/"+user.ID, user)

// Query by index - O(1) lookup
userIDs, _ := redisIndexer.QueryMultiValueIndex(
    ctx, "users", "email", "alice@example.com",
)
